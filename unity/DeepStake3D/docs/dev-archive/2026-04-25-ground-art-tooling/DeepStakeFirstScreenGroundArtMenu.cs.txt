using System.Diagnostics;
using System.IO;
using DeepStake.World;
using UnityEditor;
using UnityEditor.SceneManagement;
using UnityEngine;
using Debug = UnityEngine.Debug;

namespace DeepStake.Editor
{
    public static class DeepStakeFirstScreenGroundArtMenu
    {
        private const string ConfigAssetPath = "Assets/Resources/Configs/first-screen-ground-art-config.json";
        private const string MaterialFolderPath = "Assets/Materials/FirstScreenGroundArt";
        private const string ScreenshotScriptRelativePath = "Tools/Invoke-DeepStakeLocalScreenshotRun.ps1";

        [MenuItem("DeepStake/First Screen Art/Generate Ground Art")]
        public static void GenerateGroundArt()
        {
            var controller = FindController();
            if (controller == null)
            {
                return;
            }

            EnsureConfigFile();
            EnsurePaletteMaterials();

            var serializedController = new SerializedObject(controller);
            var zoneRoot = serializedController.FindProperty("zoneRoot")?.objectReferenceValue as Transform;
            var worldJson = serializedController.FindProperty("worldPrototypeJson")?.objectReferenceValue as TextAsset;
            if (zoneRoot == null)
            {
                Debug.LogError("[DeepStakeDev] zoneRoot is not assigned on WorldPrototype3DController.");
                return;
            }

            Undo.RegisterFullObjectHierarchyUndo(zoneRoot.gameObject, "Generate First Screen Ground Art");
            WorldPrototypeVisualPass.RefreshFirstScreenGroundArt(zoneRoot, WorldPrototype3DDefinition.FromJson(worldJson));
            EditorSceneManager.MarkSceneDirty(zoneRoot.gameObject.scene);
            Selection.activeObject = zoneRoot.gameObject;
            Debug.Log("[DeepStakeDev] First screen ground art refreshed under GeneratedGroundArt.");
        }

        [MenuItem("DeepStake/First Screen Art/Clear Generated Ground Art")]
        public static void ClearGeneratedGroundArt()
        {
            var controller = FindController();
            if (controller == null)
            {
                return;
            }

            var serializedController = new SerializedObject(controller);
            var zoneRoot = serializedController.FindProperty("zoneRoot")?.objectReferenceValue as Transform;
            if (zoneRoot == null)
            {
                Debug.LogError("[DeepStakeDev] zoneRoot is not assigned on WorldPrototype3DController.");
                return;
            }

            Undo.RegisterFullObjectHierarchyUndo(zoneRoot.gameObject, "Clear First Screen Ground Art");
            WorldPrototypeVisualPass.ClearFirstScreenGroundArt(zoneRoot);
            EditorSceneManager.MarkSceneDirty(zoneRoot.gameObject.scene);
            Selection.activeObject = zoneRoot.gameObject;
            Debug.Log("[DeepStakeDev] GeneratedGroundArt cleared.");
        }

        [MenuItem("DeepStake/First Screen Art/Capture Comparison Screenshot")]
        public static void CaptureComparisonScreenshot()
        {
            var projectRoot = Path.GetFullPath(Path.Combine(Application.dataPath, ".."));
            var scriptPath = Path.Combine(projectRoot, ScreenshotScriptRelativePath);
            if (!File.Exists(scriptPath))
            {
                Debug.LogError("[DeepStakeDev] Screenshot script not found: " + scriptPath);
                return;
            }

            var unityPath = EditorApplication.applicationPath;
            var arguments =
                "-ExecutionPolicy Bypass -File \"" + scriptPath + "\" -UnityPath \"" + unityPath + "\" -VerificationTag \"editor-ground-art\"";

            var startInfo = new ProcessStartInfo
            {
                FileName = "powershell",
                Arguments = arguments,
                WorkingDirectory = projectRoot,
                UseShellExecute = false,
                CreateNoWindow = true
            };

            Process.Start(startInfo);
            Debug.Log("[DeepStakeDev] Local screenshot workflow started. Check Pictures/Screenshot for the PNG.");
        }

        private static WorldPrototype3DController FindController()
        {
            var controller = Object.FindFirstObjectByType<WorldPrototype3DController>();
            if (controller == null)
            {
                Debug.LogError("[DeepStakeDev] Open the scene containing WorldPrototype3DController first.");
            }

            return controller;
        }

        private static void EnsurePaletteMaterials()
        {
            EnsureFolder("Assets/Materials");
            EnsureFolder(MaterialFolderPath);

            EnsureMaterial("DS_Ground_BaseSoil", new Color(0.53f, 0.45f, 0.35f));
            EnsureMaterial("DS_Ground_CompactDirtPath", new Color(0.58f, 0.49f, 0.37f));
            EnsureMaterial("DS_Ground_MixedGrassDirt", new Color(0.47f, 0.49f, 0.39f));
            EnsureMaterial("DS_Ground_RecoveryField", new Color(0.43f, 0.48f, 0.37f));
            EnsureMaterial("DS_Ground_WornStone", new Color(0.54f, 0.49f, 0.42f));
            EnsureMaterial("DS_Ground_WornConcrete", new Color(0.5f, 0.46f, 0.41f));
            EnsureMaterial("DS_Ground_Gravel", new Color(0.48f, 0.45f, 0.41f));
            EnsureMaterial("DS_Ground_NoticePlaza", new Color(0.58f, 0.5f, 0.36f));
            EnsureMaterial("DS_Ground_StorageYard", new Color(0.49f, 0.45f, 0.39f));
            EnsureMaterial("DS_Shadow_ContactSoft", new Color(0.19f, 0.16f, 0.13f), 0.85f);
            EnsureMaterial("DS_Wood_DarkWeathered", new Color(0.43f, 0.35f, 0.27f));
            EnsureMaterial("DS_Wood_LightWeathered", new Color(0.49f, 0.41f, 0.32f));
            EnsureMaterial("DS_Roof_DarkWorn", new Color(0.26f, 0.27f, 0.28f));
            EnsureMaterial("DS_Wall_MutedPlaster", new Color(0.5f, 0.47f, 0.42f));
            EnsureMaterial("DS_Wall_StoneBase", new Color(0.43f, 0.41f, 0.38f));
            EnsureMaterial("DS_Prop_DullMetal", new Color(0.42f, 0.43f, 0.44f));
            EnsureMaterial("DS_Prop_AgedSign", new Color(0.46f, 0.44f, 0.4f));
            AssetDatabase.SaveAssets();
        }

        private static void EnsureFolder(string assetPath)
        {
            if (AssetDatabase.IsValidFolder(assetPath))
            {
                return;
            }

            var parent = Path.GetDirectoryName(assetPath)?.Replace("\\", "/");
            var name = Path.GetFileName(assetPath);
            if (!string.IsNullOrWhiteSpace(parent) && !AssetDatabase.IsValidFolder(parent))
            {
                EnsureFolder(parent);
            }

            AssetDatabase.CreateFolder(parent ?? "Assets", name);
        }

        private static void EnsureMaterial(string materialName, Color color, float alpha = 1f)
        {
            var assetPath = MaterialFolderPath + "/" + materialName + ".mat";
            var material = AssetDatabase.LoadAssetAtPath<Material>(assetPath);
            if (material == null)
            {
                var shader = Shader.Find("Universal Render Pipeline/Lit");
                if (shader == null)
                {
                    shader = Shader.Find("Standard");
                }

                material = new Material(shader);
                AssetDatabase.CreateAsset(material, assetPath);
            }

            var tinted = new Color(color.r, color.g, color.b, alpha);
            if (material.HasProperty("_BaseColor"))
            {
                material.SetColor("_BaseColor", tinted);
            }
            else if (material.HasProperty("_Color"))
            {
                material.SetColor("_Color", tinted);
            }

            EditorUtility.SetDirty(material);
        }
    }
}
